#include <Wire.h>
#include <LiquidCrystal_I2C.h>

LiquidCrystal_I2C lcd(0x27, 16, 2);

void setup() {
  lcd.init();
  lcd.backlight();

  lcd.setCursor(0, 0);
  lcd.print("Arduino วัยรุ่น");

  lcd.setCursor(0, 1);
  lcd.print("Hello LCD 16x2");
}

void loop() {
}
