อ่านค่าอุณหภูมิจากเซ็นเซอร์ DHT11 แล้วแสดงผลผ่าน Serial Monitor

โค้ดสมบูรณ์

#include "DHT.h"

#define DHTPIN 2
#define DHTTYPE DHT11

DHT dht(DHTPIN, DHTTYPE);

void setup() {
  Serial.begin(9600);
  dht.begin();
}

void loop() {
  float temp = dht.readTemperature();
  Serial.print("Temperature: ");
  Serial.print(temp);
  Serial.println(" °C");
  delay(2000);
}
