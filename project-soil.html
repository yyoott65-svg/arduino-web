const int soilPin = A0;
const int relayPin = 7;

void setup() {
  pinMode(relayPin, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  int soilValue = analogRead(soilPin);
  Serial.println(soilValue);

  if (soilValue > 600) {
    digitalWrite(relayPin, HIGH);
  } else {
    digitalWrite(relayPin, LOW);
  }

  delay(1000);
}
